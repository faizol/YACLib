207 IThreadPtr Acquire(IFuncPtr func, std::size_t priority, std::string_view name, IFuncPtr acquire,
208 IFuncPtr release) final {
209 IThreadPtr t;
210 if (std::lock_guard lock{_m}; _threads.Empty()) {
211 t = new HeavyThread{};
212 ++_threads_count;
213 } else {
214 t = &_threads.PopFront();
215 }
216 static_cast<HeavyThread&>(*t).Set(priority, name, std::move(func), std::move(acquire), std::move(release));
217 return t;
218 }
219
220 void Release(IThreadPtr t) final {
221 std::unique_lock lock{_m};
222 if (_threads_count <= _cache_threads) {
223 static_cast<HeavyThread&>(*t).Wait();
224 _threads.PushFront(*t);
225 } else {
226 delete t;
227 --_threads_count;
228 }
229 }
