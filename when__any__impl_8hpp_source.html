63class AnyCombinatorBase<V, E, WhenPolicy::FirstFail> : public AnyCombinatorBase<V, E, WhenPolicy::None> {
64 using Base = AnyCombinatorBase<V, E, WhenPolicy::None>;

69 E _error;

71 };
72
73 protected:

75 : Base{size, std::move(core)}, _state{ResultState::Empty}, _unit{} {
76 }
77
78 ~AnyCombinatorBase() {
79 auto done = Base::_done.load(std::memory_order_acquire);
80 auto state = _state.load(std::memory_order_acquire);
81 if (state == ResultState::Error) {
82 if (!done) {
83 Base::_core->Set(std::move(_error));
84 }
85 _error.~E();
86 } else if (state == ResultState::Exception) {
87 if (!done) {
88 Base::_core->Set(std::move(_exception));
89 }
90 _exception.~exception_ptr();
91 }
92 }
93
94 void Combine(Result<V, E>&& result) {
95 auto state = result.State();
96 if (state == ResultState::Value) {
97 return Base::Combine(std::move(result));
98 }
99 auto old_state = _state.load(std::memory_order_acquire);
100 if (old_state == ResultState::Empty &&
101 _state.compare_exchange_strong(old_state, state, std::memory_order_acq_rel, std::memory_order_relaxed)) {
102 if (state == ResultState::Exception) {
103 new (&_exception) std::exception_ptr{std::move(result).Exception()};
104 } else {
105 YACLIB_ERROR(state != ResultState::Error, "state should be Error, but here it's Empty");
106 new (&_error) E{std::move(result).Error()};
107 }
108 }
109 }
