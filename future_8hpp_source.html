183template <typename V = Unit, typename E = StopError, typename... Args>
184auto MakeFuture(Args&&... args) {
185 if constexpr (sizeof...(Args) == 0) {
186 return Future{MakeIntrusive<detail::ResultCore<void, E>>(Unit{})};
187 } else if constexpr (sizeof...(Args) == 1) {
188 using T = std::conditional_t<std::is_same_v<V, Unit>, head_t<Args...>, V>;
189 static_assert(!std::is_same_v<T, Unit>);
190 return Future{MakeIntrusive<detail::ResultCore<T, E>>(std::forward<Args>(args)...)};
191 } else {
192 static_assert(!std::is_same_v<V, Unit>);
193 return Future{MakeIntrusive<detail::ResultCore<V, E>>(std::forward<Args>(args)...)};
194 }
