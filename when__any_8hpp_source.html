18 *
19 * \tparam P policy WhenAny errors
20 * \tparam It type of passed iterator
21 * \tparam T type of all passed futures
22 * \param begin , size the range of futures to combine
23 * \return Future<T>
24 */
25template <WhenPolicy P = WhenPolicy::LastFail, typename It, typename T = typename std::iterator_traits<It>::value_type>
26auto WhenAny(It begin, std::size_t size) {
27 static_assert(is_future_v<T>, "WhenAny function Iterator must be point to some Future");
28 using V = future_value_t<T>;
29 using E = future_error_t<T>;
30 if (size == 0) {
31 return Future<V, E>{};
32 }
33 if (size == 1) {
34 return std::move(*begin);
35 }
36 auto [future, combinator] = detail::AnyCombinator<V, E, P>::Make(size);
37 detail::WhenImpl(combinator, begin, size);
38 return std::move(future);
39}
