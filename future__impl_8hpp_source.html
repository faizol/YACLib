67 template <typename T>
68 void Call(Core& self, T&& r) noexcept {
69 try {
70 constexpr bool kIsRun = std::is_same_v<T, Unit>;
71 static_assert(kIsRun || std::is_same_v<T, Result<Arg, E>>);
72 if constexpr (kIsRun || is_invocable_v<Invoke, Result<Arg, E>>) {
73 CallResolveAsync(self, std::forward<T>(r));
74 } else {
75 CallResolveState(self, std::forward<T>(r));
76 }
77 } catch (...) {
78 self.Set(std::current_exception());
79 }
80 }
81
82 private:
83 void CallResolveState(Core& self, Result<Arg, E>&& r) {
84 const auto state = r.State();
85 if constexpr (is_invocable_v<Invoke, Arg>) {
86 if (state == ResultState::Value) {
87 CallResolveAsync(self, std::move(r).Value());
88 } else if (state == ResultState::Error) {
89 self.Set(std::move(r).Error());
90 } else {
91 assert(state == ResultState::Exception);
92 self.Set(std::move(r).Exception());
93 }
94 } else {
95 /**
96 * We can't use this strategy for other Functor::Invoke,
97 * because in that case user will not have compile error for that case:
98 * MakeFuture<void>() // state == ResultState::Value
99 * .ThenInline([](E/std::exception_ptr) -> yaclib::Result/Future<double> {
100 * throw std::runtime_error{""};
101 * }).ThenInline([](yaclib::Result<double>) { // need double value, we only have void
102 * return 1;
103 * });
104 * TLDR: Before and after the "recovery" callback must have the same value type
105 */
106 constexpr bool kIsError = is_invocable_v<Invoke, E>;
107 constexpr bool kIsException = is_invocable_v<Invoke, std::exception_ptr>;
108 static_assert(kIsError ^ kIsException, "Recovery callback should be invokable with E or std::exception_ptr");
109 constexpr auto kState = kIsError ? ResultState::Error : ResultState::Exception;
110 if (state == kState) {
111 using T = std::conditional_t<kIsError, E, std::exception_ptr>;
112 CallResolveAsync(self, std::move(r).template Extract<T>());
113 } else {
114 self.Set(std::move(r));
115 }
116 }
117 }

119 template <typename T>
120 void CallResolveAsync(Core& self, T&& value) {
121 if constexpr (Async) {
122 auto future = CallResolveVoid(std::forward<T>(value));
123 auto core = std::exchange(future.GetCore(), nullptr);
124 self.IncRef();
125 core->SetCallbackInline(self, true);
126 } else {
127 self.Set(CallResolveVoid(std::forward<T>(value)));
128 }
129 }

131 template <typename T>
132 auto CallResolveVoid(T&& value) {
133 constexpr bool kArgVoid = std::is_same_v<T, Unit>;
134 constexpr bool kRetVoid = std::is_void_v<invoke_t<Invoke, std::conditional_t<kArgVoid, void, T>>>;
135 if constexpr (kRetVoid) {
136 if constexpr (kArgVoid) {
137 std::forward<Invoke>(_f)();
138 } else {
139 std::forward<Invoke>(_f)(std::forward<T>(value));
140 }
141 return Unit{};
142 } else if constexpr (kArgVoid) {
143 return std::forward<Invoke>(_f)();
144 } else {
145 return std::forward<Invoke>(_f)(std::forward<T>(value));
146 }
147 }

149 Store _f;

152template <CoreType Type, typename Arg, typename E, typename Functor>
153auto* MakeCore(Functor&& f) {
154 static_assert(Type != CoreType::Run || std::is_void_v<Arg>, "It makes no sense to receive some value in first step");
155 using AsyncRet = result_value_t<typename detail::Return<Arg, E, Functor>::Type>;
156 static_assert(Type != CoreType::Subscribe || std::is_void_v<AsyncRet>,
157 "It makes no sense to return some value in Subscribe, since no one will be able to use it");
158 using Ret = result_value_t<future_value_t<AsyncRet>>;
159 constexpr bool kIsAsync = is_future_v<AsyncRet>;
160 using Wrapper =
161 detail::FunctorWrapper<Type == CoreType::Subscribe, kIsAsync, Ret, Arg, E, decltype(std::forward<Functor>(f))>;
162 // TODO(MBkkt) Think about inline/subscribe optimization
163 using Core = detail::Core<Ret, Arg, E, Wrapper, Type>;
164 constexpr std::size_t kRef = 1 + static_cast<std::size_t>(Type == CoreType::Then);
165 return new detail::AtomicCounter<Core>{kRef, std::forward<Functor>(f)};
166}

168template <CoreType Type, bool Inline, typename Arg, typename E, typename Functor>
169auto SetCallback(ResultCorePtr<Arg, E> caller, Functor&& f) {
170 static_assert(Type != CoreType::Run);
