22template <typename Event, typename Timeout, typename Range>
23bool WaitRange(const Timeout& timeout, const Range& range, std::size_t count) {
24 AtomicCounter<Event, EventDeleter> event{count + 1};
25 // event ref counter = n + 1, it is optimization: we don't want to notify when return true immediately
26 auto const wait_count = range([&](detail::BaseCore& core) {
27 return core.Empty() && core.SetWait(event);
28 });
29 if (wait_count == 0 || event.SubEqual(count - wait_count + 1)) {
30 return true;
31 }
32 auto token = event.Make();
33 std::size_t reset_count = 0;
34 if constexpr (!std::is_same_v<Timeout, NoTimeoutTag>) {
35 // If you have problem with TSAN here, check this link: https://github.com/google/sanitizers/issues/1259
36 // TLDR: new pthread function is not supported by thread sanitizer yet.
37 if (event.Wait(token, timeout)) {
38 return true;
39 }
40 reset_count = range([](detail::BaseCore& core) {
41 return core.ResetWait();
42 });
43 if (reset_count != 0 && (reset_count == wait_count || event.SubEqual(reset_count))) {
44 return false;
45 }
46 // We know we have `wait_count - reset_count` Results, but we must wait until event was not used by cores
47 }
48 event.Wait(token);
49 return reset_count == 0;
50}

52template <typename Event, typename Timeout, typename... Cores>
53bool WaitCore(const Timeout& timeout, Cores&... cores) {
54 static_assert(sizeof...(cores) >= 1, "Number of futures must be at least one");
55 static_assert((... && std::is_same_v<detail::BaseCore, Cores>), "Futures must be Future in Wait function");
56 auto range = [&](auto&& functor) {
57 return (... + static_cast<std::size_t>(functor(cores)));
58 };
59 return WaitRange<Event>(timeout, range, sizeof...(cores));
60}

62template <typename Event, typename Timeout, typename Iterator>
63bool WaitIterator(const Timeout& timeout, Iterator it, std::size_t size) {
64 static_assert(is_future_v<typename std::iterator_traits<Iterator>::value_type>,
65 "Wait function Iterator must be point to some Future");
66 if (size == 0) {
67 return true;
68 }
69 auto range = [&](auto&& functor) {
70 auto temp_it = it;
71 std::size_t count = 0;
72 for (std::size_t i = 0; i != size; ++i) {
73 count += static_cast<std::size_t>(functor(*temp_it->GetCore()));
74 ++temp_it;
75 }
76 return count;
77 };
78 return WaitRange<Event>(timeout, range, size);
79}
